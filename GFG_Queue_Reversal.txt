//Function to reverse the queue.
class Solution
{
    public:
    queue<int> rev(queue<int> q)
    {
        // add code here.
        stack<int> st;
        int x = q.size();
        while(x>0){
            int y = q.front();
            st.push(y);
            q.pop();
            x--;
        }
        x = st.size();
        while(x>0){
            int y = st.top();
            q.push(y);
            st.pop();
            x--;
        }
        return q;
    }
};
